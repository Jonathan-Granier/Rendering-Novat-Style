121 void Shader::initialize()
122 {
123 
124  // 1. retrieve the vertex/fragment source code from filePath
125  string vertexCode, fragmentCode;
126  ifstream vShaderFile, fShaderFile;
127  // ensure ifstream objects can throw exceptions:
128  vShaderFile.exceptions (ifstream::failbit | ifstream::badbit);
129  fShaderFile.exceptions (ifstream::failbit | ifstream::badbit);
130  try
131  {
132  // open files
133  vShaderFile.open(_vertexPath);
134  fShaderFile.open(_fragmentPath);
135  stringstream vShaderStream, fShaderStream;
136  // read file's buffer contents into streams
137  vShaderStream << vShaderFile.rdbuf();
138  fShaderStream << fShaderFile.rdbuf();
139  // close file handlers
140  vShaderFile.close();
141  fShaderFile.close();
142  // convert stream into string
143  vertexCode = vShaderStream.str();
144  fragmentCode = fShaderStream.str();
145  }
146  catch(ifstream::failure e)
147  {
148  cerr << "ERROR::SHADER::FILE_NOT_SUCCESFULLY_READ" << endl;
149  }
150  const char* vShaderCode = vertexCode.c_str();
151  const char* fShaderCode = fragmentCode.c_str();
152 
153 
154 
155  // 2. compile shadersfragmentPaht
156  unsigned int vertex, fragment;
157 
158  // vertex Shader
159  vertex = glCreateShader(GL_VERTEX_SHADER);
160  glShaderSource(vertex, 1, &vShaderCode, NULL);
161  glCompileShader(vertex);
162  checkCompileErrors(vertex,"VERTEX",_name);
163 
164  //Fragment Shader
165  fragment = glCreateShader(GL_FRAGMENT_SHADER);
166  glShaderSource(fragment, 1, &fShaderCode, NULL);
167  glCompileShader(fragment);
168  // print compile errors if any
169  checkCompileErrors(fragment,"FRAGMENT",_name);
170 
171  // shader Program
172  _id = glCreateProgram();
173  glAttachShader(_id, vertex);
174  glAttachShader(_id, fragment);
175  glLinkProgram(_id);
176  // print linking errors if any
177  checkCompileErrors(_id,"PROGRAM",_name);
178 }

181 {
182  GLint success;

185  {
186  glGetShaderiv(shader, GL_COMPILE_STATUS, &success);

191  }
