140 void Scene::generateLaplacienPyramid(){
141  if(_doReloadLaplacienPyramid){
142 
143  // The gauss Blur
144  bool firstMap = true;
145  unsigned int previousI = 0;
146  for(unsigned int i = 0; i < _scalesManager.size(); i++)
147  {
148  if(_scalesManager[i]->enabled){
149  if(!firstMap)
150  _scalesManager[i]->scale->generateHeightMap(_scalesManager[previousI]->scale->getHeightMap());
151  previousI = i;
152  firstMap = false;
153  }
154  }
155  // The difference
156  bool lastMap = true;
157  shared_ptr<Texture> nextHeightMap;
158  for(int i = _scalesManager.size()-1; i >=0 ; i--){
159  if(_scalesManager[i]->enabled){
160  _scalesManager[i]->scale->generateEditHeightMap(nextHeightMap,lastMap);
161  lastMap = false;
162  _scalesManager[i]->scale->generateNormalMap();
163  _scalesManager[i]->scale->generateSlantMap();
164  nextHeightMap = _scalesManager[i]->scale->getHeightMap();
165  }
166  }
167  _doReloadLaplacienPyramid = false;
168 
169  }
